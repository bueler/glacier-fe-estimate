\begin{frame}{the free-boundary problem for glacier surface elevation}

\begin{columns}
\begin{column}{0.75\textwidth}

\begin{itemize}
\item surface kinematical equation (SKE) in glaciology:
   $$\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a = 0$$
\item a time-dependent, nonlinear complementarity problem (NCP) gives the \emph{free-boundary} meaning of the SKE:

\vspace{-6mm}

\begin{align*}
s - b &\ge 0 &&\phantom{x} \\
\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a &\ge 0 \\
(s - b) \left(\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a\right) &= 0
\end{align*}

\vspace{-2mm}
    \begin{itemize}
    \item[$\circ$] applies everywhere in $[0,T]\times \Omega$
    \item[$\circ$] applies for any ice dynamics model
    \item[$\circ$] appears first in (Calvo et al 2003), but only for shallow ice
    \end{itemize}
\end{itemize}

\end{column}
\begin{column}{0.30\textwidth}
\includegraphics[width=\textwidth]{mapplane}

\vspace{10mm}
\hfill \includegraphics[width=0.9\textwidth]{xtcrop}
\end{column}
\end{columns}
\end{frame}


\begin{frame}{Stokes is the dynamical model within the ice}

\hfill \includegraphics[width=0.55\textwidth]{stokesdomainOmega}

\vspace{-5mm}

\begin{itemize}
\item fix $t$ and define:
    $$\alert{\Lambda(s) = \{(x,z)\,:\,b(x)<z<s(t,x)\}} \qquad \subset \RR^3$$
\item Glen-Stokes equations \alert{in $\Lambda(s)$} with $\pp = (\nn + 1)/\nn \approx 4/3$:
\begin{align*}
- \nabla \cdot \left(2 \nu(D\bu)\, D\bu\right) + \nabla p &= \rhoi \bg  \\
\nabla \cdot \bu &= 0  \\
\nu(D\bu) &= \nu_0 |D\bu|^{\pp-2} && \quad {\scriptsize \leftarrow \text{needs regularization?}}
\end{align*}

\vspace{-1mm}
\item boundary conditions:
\begin{align*}
&\left(2 \nu(D\bu) D\bu - pI\right) \bn_s = \bzero   && \where{on $\Gamma_s \subset \partial\Lambda(s)$} \\
&\bu  = \bzero \text{ } \,\text{or } \,f(\bu,D\bu)=0 && \where{on $\Gamma_b \subset \partial\Lambda(s)$}
\end{align*}
\end{itemize}
\end{frame}


\begin{frame}{the standard model for glacier evolution}

\begin{align*}
s - b &\ge 0 & &\where{in $\Omega$} \\
\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a &\ge 0 \\
(s - b) \left(\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a\right) &= 0 \\
- \nabla \cdot \left(2 \nu_0 |D\bu|^{\pp-2}\, D\bu\right) + \nabla p &= \rhoi \bg && \where{in $\Lambda(s)$} \\
\nabla \cdot \bu &= 0 \\
\left(2 \nu(D\bu) D\bu - pI\right) \bn_s &= \bzero && \where{on $\Gamma_s \subset \partial\Lambda(s)$} \\
\bu  = \bzero \quad \text{or} \quad f(\bu,D\bu) &= 0 && \where{on $\Gamma_b \subset \partial\Lambda(s)$}
\end{align*}

\bigskip

\only<1>{
\begin{smodel}[a suggested name for the system above!]
an \alert{NCP} in $[0,T]\times \Omega$, for fixed domain $\Omega \subset \RR^2$, \alert{coupled to Stokes within the ice}, in a time-dependent domain $\Lambda(s) \subset \RR^3$
\end{smodel}
}
\only<2>{
\begin{smodel}[as a dynamical system]
an inequality-constrained \alert{differential algebraic equations} (DAE) system in $\infty$ dimensions
\end{smodel}
}
\end{frame}


\begin{frame}{the standard model wants implicit time-stepping}

\begin{center}
\includegraphics[width=0.5\textwidth]{implicitstep}
\end{center}

\begin{itemize}
\item DAE systems are stiff so implicit time discretization is expected
    \begin{itemize}
    \item[$\circ$] stiff decay BDF schemes, or Radau schemes
    \end{itemize}
\item for example, consider backward Euler on SKE $\frac{\partial s}{\partial t} - \bu|_s \cdot \bn_s - a = 0$:
    $$\frac{s^k - s^{k-1}}{\Delta t} - \bu|_{s^k} \cdot \bn_{s^k} - \frac{1}{\Delta t}\int_{t_{k-1}}^{t_k} a(t)\,dt = 0$$
\end{itemize}
\end{frame}


\begin{frame}{implicit time-step problem is a variational inequality (VI)}

\begin{itemize}
\item VI = (weak form of an NCP)
\item let $\cX$ denote a Banach space of elevation functions on $\Omega$
\item $b\in\cX$
\item for $\Delta t > 0$ define:

\vspace{-5mm}
\begin{align*}
\cK &= \{r \in \cX\,:\, r \ge b \,\text{ a.e. } \Omega\} \\
\ell^k(x) &= s^{k-1}(x) + \int_{t_{k-1}}^{t_k} a(t,x)\,dt \\
F(s)[q] &= \int_\Omega (s - \Delta t\,\bu|_s \cdot \bn_s) q  \hspace{15mm} \text{{\scriptsize (note \, $F:\cK\to\cX'$)}}
\end{align*}
\end{itemize}

\medskip
\begin{block}{definition\footnote{{\scriptsize B. \emph{Surface elevation errors in finite element Stokes models for glacier evolution}, \href{https://www.arxiv.org/abs/2408.06470}{arxiv:2408.06470}}}}
the weak form \emph{backward Euler time-step problem}, for $t_k \in [0,T]$, is to find the new surface elevation $s^k \approx s(t_k,x) \in \cK$ solving the VI
$$F(s^k)[r-s^k] \ge \ell^k[r-s^k] \quad \text{for all } r \in \cK$$
\end{block}

\medskip
\end{frame}


\begin{frame}{implicit time-step problem is a variational inequality (VI)}


\begin{block}{the main idea}
the weak form \emph{backward Euler time-step problem}
    $$F(s^k)[r-s^k] \ge \ell^k[r-s^k] \quad \text{for all } r \in \cK$$
\alert{simultaneously} computes
\begin{itemize}
\item the new map-plane domain covered by the glacier,
\item the new surface elevation $s^k$ solving the SKE, and
\item the new Stokes solution $\bu^k,p^k$ on $\Lambda(s^k) = \{b<z<s^k\} \subset \RR^3$
\end{itemize}
\end{block}

\begin{itemize}
\item a key mechanism is the \alert{surface motion map}, from $s$ to the surface motion term in the SKE, using the solution of the stokes problem:
	$$s \mapsto \bu_s\cdot \bn_s$$
\end{itemize}
\end{frame}

\section{well-posedness?}

\begin{frame}{why would we care about well-posedness?}

\begin{itemize}
\item (maybe you came for the finite elements?)
\item recall the canonical story about convergence of the FEM, in which \alert{$u$} denotes the continuum solution:
{\small
\begin{align*}
-\grad^2 \alert{u} &= f \\
&\downarrow & &\text{\emph{to weak form}} \\
a(\alert{u},v) &= \ell_f(v) \\
&\downarrow & &\text{\emph{finite elements (FE)}} \\
a(u_h,v_h) &= \ell_f(v_h) \\
&\downarrow & &\text{\emph{Cea's lemma}} \\
\|u_h-\alert{u}\| &\le C \inf_{v_h} \|v_h-\alert{u}\| \\
&\downarrow & &\text{\emph{interpolation theory}} \\
\|u_h-\alert{u}\| &\le C h^q \|\alert{u}\|_{H^r}
\end{align*}
}
\item to tell this FEM story for glacier surface elevations, well-posedness must be either proven, conjectured, or politely ignored
\item this talk chooses ``conjectured'' \dots because I don't know a proof
\end{itemize}
\end{frame}


\begin{frame}{well-posedness is only conjectural}

\begin{conjecture}
There exists a $\qq>2$ so that in the space $\cX = W^{1,\qq}(\Omega)$, for sufficiently small $\Delta t>0$, the backward Euler time-step VI problem
    $$F(s^k)[r-s^k] \ge \ell^k[r-s^k] \quad \text{for all } r \in \cK,$$
is well-posed for the new surface elevation $s^k\in\cK$ in the admissible subset $\cK = \{r \ge b\} \subset \cX$.
\end{conjecture}

\section{finite element approximation}

\begin{frame}{FE method}

\begin{center}
\includegraphics[width=0.7\textwidth]{extruded}
\end{center}

\begin{cproblem}
find $s\in\cK = \{r \ge b\}$ which solves the backward-Euler step VI:
   $$F(s)[r-s] \ge \ell^k[r-s] \quad \text{for all } r \in \cK$$
\end{cproblem}

\begin{itemize}
\item proposed FE spaces:
   \begin{itemize}
   \item[$\circ$] $b_h,s_h \in P_1$
   \item[$\circ$] $\bu_h,p_h \in P_2 \times P_1$, on an extruded mesh \hfill {\scriptsize $\leftarrow$ one possibility}
   \end{itemize}
\item FE method for $s_h\in\cK_h = \{r_h \ge b_h\}$:
   $$F_h(s_h)[r_h-s_h] \ge \ell^k[r_h-s_h] \quad \text{for all } r_h \in \cK_h$$

   \begin{itemize}
   \item[$\circ$] mostly conforming, especially if one pretends $b_h=b$
   \end{itemize}
\end{itemize}

\begin{frame}{error theorem for a backward Euler step}

\noindent \includegraphics[width=0.6\textwidth]{extruded} \hfill \includegraphics[width=0.23\textwidth]{mapplane}

\medskip
\begin{theorem}[B '24]
Make assumptions sufficient for well-posedness.  Let $\Omega_A(s)$ be  the exact active set, and $\Pi_h$ the interpolation into $\cK_h = \{r_h \ge b_h\}$.  Then the FE error in the new ($t_k$) surface elevation is bounded by 3 terms:
\begin{align*}
\|s_h-s\|_{W^{1,\qq}}^\qq &\le \, \frac{c_1}{\Delta t} \int_{\Omega_A(s)} (b - \ell^k) (b_h - b) \\
   &\quad\, + C(s_h) \big\|\bu_h - \bu\big\|_{W^{1,\pp}} \\
   &\quad\, + c_0 \|\Pi_h(s) - s\|_{W^{1,\qq}}^{\qq'}
\end{align*}
\end{theorem}

\noindent \emph{Proof.} Make the Falk (1974) technique for VIs fully nonlinear. \qed
\end{frame}

\begin{frame}{summary}
\begin{itemize}
\item implicit time-stepping is appropriate for geometry-evolving, Stokes models for glaciers
    \begin{itemize}
    \item[$\circ$] both a differential-algebraic system and a free-boundary problem
    \end{itemize}
\item \textbf{conjecture.} a regularization of the surface motion $-\bu|_s\cdot \bn_s$, for $\bu$ solving the Stokes problem, is coercive over admissible $s\in W^{1,\qq}(\Omega)$
\item \textbf{theorem.} supposing this coercivity, each continuous-space, backward Euler time step problem is well-posed
\item \textbf{theorem.} assuming well-posedness, the surface elevation FE error is bounded by a sum of 3 terms:
    \begin{enumerate}
    \item discretizing the bed elevation ($b_h$ versus $b$)
    \item numerically solving the Stokes equations ($\bu_h$ versus $\bu$)
    \item Cea's lemma for the surface elevation ($s_h$ versus $\Pi_h(s)$) \strut
    \end{enumerate}
\end{itemize}
